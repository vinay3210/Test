try
    -- Prompt the user (Equivalent to VBScript InputBox)
    set dialogResult to display dialog "Hello! How long do you want to keep your system awake?" & return & return & "Enter minutes:" default answer "" with title "Awake Duration"
    set inputMinutes to text returned of dialogResult
    
    -- Attempt to cast the input to an integer to validate it
    set numMinutes to inputMinutes as integer
    
    if numMinutes > 0 then
        -- Convert minutes to seconds for the caffeinate command
        set numSeconds to numMinutes * 60
        
        -- Run the macOS caffeinate command. 
        -- This will pause the script exactly like your WScript.Sleep loop
        do shell script "caffeinate -t " & numSeconds
        
        -- Success message (Equivalent to VBScript MsgBox)
        display dialog "Forced awake time over. Back to normal routine." with title "Task Completed" buttons {"OK"} default button "OK" with icon note
    else
        -- Force an error to jump to the error handling block
        error "Must be greater than zero" 
    end if
    
on error errMsg errNum
    -- Error number -128 means the user clicked "Cancel". We do nothing.
    -- If it's any other error (like typing text instead of numbers), show the error dialog.
    if errNum is not -128 then
        display dialog "Incorrect input, script won't run." & return & "You can only enter a numeric value greater than zero." with title "Task Failed" buttons {"OK"} default button "OK" with icon stop
    end if
end try
